    Extend `resolveJobModel` from Plan 01 (currently returns just `job.model ?? undefined`):
    ```typescript
    export async function resolveJobModel(job: Job): Promise<string | undefined> {
      if (job.model) return job.model.trim().toLowerCase();
      if (job.agent) {
        try {
          const agent = await loadAgent(job.agent);
          if (agent?.defaultModel) return agent.defaultModel.trim().toLowerCase();
        } catch {
          // Agent load failure should not break cron — fall through to daemon default
        }
      }
      return undefined;
    }
    ```
